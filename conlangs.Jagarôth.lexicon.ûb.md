-ûb :: combinative form of tûb ::
tûb :: nominal :: season, instance, moment, occasion
also
tûb :: adverb :: at a/the time
- example: tûb maimisûr mûmburnûr-thor :: in a time of madness and death

l- :: interrogative marker
lûb :: at what time? when?
tik :: demonstrative :: this (near speaker)
tak :: demonstrative :: that (near interlocutor)
tok :: demonstrative :: that one over yonder (from from both)

tikûb :: at this time
takûb :: at that time
tokûb :: at that distant time

kek :: adj :: other
kekûb :: at another time
